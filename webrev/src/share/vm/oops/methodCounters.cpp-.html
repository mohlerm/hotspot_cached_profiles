  28 MethodCounters* MethodCounters::allocate(methodHandle mh, TRAPS) {
  29   ClassLoaderData* loader_data = mh->method_holder()->class_loader_data();
  30   return new(loader_data, size(), false, MetaspaceObj::MethodCountersType, THREAD) MethodCounters(mh);
  31 }
  32 
  33 void MethodCounters::clear_counters() {
  34   invocation_counter()->reset();
  35   backedge_counter()->reset();
  36   set_interpreter_throwout_count(0);
  37   set_interpreter_invocation_count(0);
  38   set_nmethod_age(INT_MAX);
  39 #ifdef TIERED
  40   set_prev_time(0);
  41   set_rate(0);
  42   set_highest_comp_level(0);
  43   set_highest_osr_comp_level(0);
  44 #endif
  45 }
  46 
  47 
  48 int MethodCounters::highest_comp_level() const {
  49 #ifdef TIERED
  50   return _highest_comp_level;
  51 #else
  52   return CompLevel_none;
  53 #endif
  54 }
  55 
  56 void MethodCounters::set_highest_comp_level(int level) {
  57 #ifdef TIERED
  58   _highest_comp_level = level;
  59 #endif
  60 }
  61 
  62 int MethodCounters::highest_osr_comp_level() const {

  65 #else
  66   return CompLevel_none;
  67 #endif
  68 }
  69 
  70 void MethodCounters::set_highest_osr_comp_level(int level) {
  71 #ifdef TIERED
